import { Component } from '@angular/core';

@Component({
  selector: 'app-schedule-time',
  templateUrl: './schedule-time.component.html',
  styleUrls: ['./schedule-time.component.css']
})
export class ScheduleTimeComponent {

  hour: string[] = [];
  newHour: string = '';
  dia: string[] = [];

  constructor() { }

  ngOnInit(): void {
  }

  addHour() {
    if (this.newHour.trim() !== '') {
      this.hour.push(this.newHour);
      this.newHour = '';
    }
  }

  diasDaSemana = [
    { name: 'DOM', checked: false, hour: '' },
    { name: 'SEG', checked: false, hour: '' },
    { name: 'TER', checked: false, hour: '' },
    { name: 'QUA', checked: false, hour: '' },
    { name: 'QUI', checked: false, hour: '' },
    { name: 'SEX', checked: false, hour: '' },
    { name: 'SÁB', checked: false, horhoura: '' }
  ];

  atualizarHora(dia: any) {
    if (dia.checked) {
      // Lógica para definir a hora para este dia, por exemplo:
      dia.hora = new Date().toLocaleTimeString();
    } else {
      dia.hora = ''; // Limpar a hora se o checkbox for desmarcado.
    }
  }

}
